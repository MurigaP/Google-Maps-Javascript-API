GMaps.js — Static map

You can use the same styles as for a normal styled map to make a static map using GMaps.js:

url = GMaps.staticMapURL({
  size: [610, 350],
  lat: -12.043333,
  lng: -77.028333
  styles: [{
    featureType: "all",
    elementType: "all",
    stylers: [
      { hue: '#FF0000' }
    ]
  },
  {
    featureType: "landscape",
    elementType: "all",
    stylers: [
      { hue: '#0A2D3F'},
      { saturation: 50 },
      { lightness: -10 }
    ]
  }]
});

$('<img/>').attr('src', url)
  .appendTo('#map');
